        function loadQuestions() {

            fetch('reproductive_physiology_question_bank.json')

                .then(response => {

                    if (!response.ok) {

                        throw new Error('Failed to load questions');

                    }

                    return response.json();

                })

                .then(data => {

                    allQuestions = data.sections.multiple_choice_questions;

                    document.getElementById('loadingScreen').style.display = 'none';

                    document.getElementById('startScreen').style.display = 'block';

                })

                .catch(error => {

                    console.error('Error loading questions:', error);

                    document.getElementById('loadingScreen').style.display = 'none';

                    document.getElementById('errorScreen').style.display = 'block';

                });

        }

        function shuffleArray(array) {

            const newArray = [...array];

            for (let i = newArray.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [newArray[i], newArray[j]] = [newArray[j], newArray[i]];

            }

            return newArray;

        }

        function startQuiz() {

            quizMode = 'full';

            questions = shuffleArray(allQuestions);

            userAnswers = new Array(questions.length).fill(null);

            currentQuestionIndex = 0;

            score = 0;

            document.getElementById('startScreen').style.display = 'none';

            document.getElementById('quizScreen').style.display = 'block';

            showQuestion();

        }

        function startCustomQuiz() {

            quizMode = 'custom';

            questions = shuffleArray(allQuestions).slice(0, 20);

            userAnswers = new Array(questions.length).fill(null);

            currentQuestionIndex = 0;

            score = 0;

            document.getElementById('startScreen').style.display = 'none';

            document.getElementById('quizScreen').style.display = 'block';

            showQuestion();

        }

        function showQuestion() {

            const question = questions[currentQuestionIndex];

            // Update progress

            const progress = ((currentQuestionIndex + 1) / questions.length) * 100;

            document.getElementById('progressBar').style.width = progress + '%';

            // Update question info

            document.getElementById('questionCounter').textContent =

                `Question ${currentQuestionIndex + 1} of ${questions.length}`;

            document.getElementById('topicBadge').textContent = question.topic.replace('Topic ', '');

            // Update question text

            document.getElementById('questionText').textContent = question.question;

            // Create options

            const optionsContainer = document.getElementById('optionsContainer');

            optionsContainer.innerHTML = '';

            const optionLetters = ['A', 'B', 'C', 'D', 'E'];

            const userAnswer = userAnswers[currentQuestionIndex];

            optionLetters.forEach(letter => {

                if (question.options[letter]) {

                    const btn = document.createElement('button');

                    btn.className = 'option-btn';

                    // Show correct/incorrect styling if already answered

                    if (userAnswer) {

                        if (letter === question.correct_answer) {

                            btn.classList.add('correct');

                        } else if (letter === userAnswer.selected && userAnswer.selected !== question.correct_answer) {

                            btn.classList.add('incorrect');

                        }

                        btn.disabled = true;

                    }

                    btn.innerHTML = `<span class="option-letter">${letter}</span>${question.options[letter]}`;

                    btn.onclick = () => selectOption(letter);

                    optionsContainer.appendChild(btn);

                }

            });

            // Show feedback if already answered

            const feedback = document.getElementById('feedback');

            if (userAnswer) {

                feedback.style.display = 'block';

                if (userAnswer.isCorrect) {

                    feedback.className = 'feedback correct';

                    feedback.innerHTML = '✓ Correct! Well done.';

                } else {

                    feedback.className = 'feedback incorrect';

                    feedback.innerHTML = `✗ Incorrect. The correct answer is <strong>${question.correct_answer}</strong>.`;

                }

                document.getElementById('nextBtn').disabled = false;

            } else {

                feedback.style.display = 'none';

                document.getElementById('nextBtn').disabled = true;

            }

            // Update navigation buttons

            document.getElementById('prevBtn').style.visibility =

                currentQuestionIndex > 0 ? 'visible' : 'hidden';

            document.getElementById('nextBtn').textContent =

                currentQuestionIndex === questions.length - 1 ? 'Finish Quiz' : 'Next Question';

        }

        function selectOption(selectedLetter) {

            if (userAnswers[currentQuestionIndex]) return; // Already answered

            const question = questions[currentQuestionIndex];

            const isCorrect = selectedLetter === question.correct_answer;

            userAnswers[currentQuestionIndex] = {

                selected: selectedLetter,

                isCorrect: isCorrect

            };

            if (isCorrect) score++;

            // Update UI

            const buttons = document.querySelectorAll('.option-btn');

            const optionLetters = ['A', 'B', 'C', 'D', 'E'];

            buttons.forEach((btn, index) => {

                const letter = optionLetters[index];

                btn.disabled = true;

                if (letter === question.correct_answer) {

                    btn.classList.add('correct');

                } else if (letter === selectedLetter && !isCorrect) {

                    btn.classList.add('incorrect');

                }

            });

            // Show feedback

            const feedback = document.getElementById('feedback');

            feedback.style.display = 'block';

            if (isCorrect) {

                feedback.className = 'feedback correct';

                feedback.innerHTML = '✓ Correct! Well done.';

            } else {

                feedback.className = 'feedback incorrect';

                feedback.innerHTML = `✗ Incorrect. The correct answer is <strong>${question.correct_answer}</strong>.`;

            }

            document.getElementById('nextBtn').disabled = false;

        }

        function nextQuestion() {

            if (currentQuestionIndex < questions.length - 1) {

                currentQuestionIndex++;

                showQuestion();

            } else {

                showResults();

            }

        }

        function previousQuestion() {

            if (currentQuestionIndex > 0) {

                currentQuestionIndex--;

                showQuestion();

            }

        }

        function showResults() {

            document.getElementById('quizScreen').style.display = 'none';

            document.getElementById('resultsScreen').style.display = 'block';

            const percentage = Math.round((score / questions.length) * 100);

            const incorrect = questions.length - score;

            document.getElementById('scoreNumber').textContent = percentage + '%';

            document.getElementById('totalQuestions').textContent = questions.length;

            document.getElementById('correctAnswers').textContent = score;

            document.getElementById('incorrectAnswers').textContent = incorrect;

            document.getElementById('accuracyPercent').textContent = percentage + '%';

            // Show review of incorrect answers

            const reviewContainer = document.getElementById('reviewContainer');

            const incorrectAnswers = userAnswers.map((answer, index) => ({...answer, index}))

                .filter(answer => !answer.isCorrect);

            if (incorrectAnswers.length > 0) {

                document.getElementById('reviewSection').style.display = 'block';

                reviewContainer.innerHTML = '';

                incorrectAnswers.forEach(answer => {

                    const question = questions[answer.index];

                    const div = document.createElement('div');

                    div.className = 'review-item';

                    div.innerHTML = `

                        <div class="review-question">Q${answer.index + 1}: ${question.question}</div>

                        <div class="review-answer">

                            Your answer: <strong>${answer.selected}</strong> |

                            Correct answer: <strong>${question.correct_answer}</strong>

                        </div>

                    `;

                    reviewContainer.appendChild(div);

                });

            }

        }

            if (quizMode === 'custom') {

            } else {

                startQuiz();

            }

        }
